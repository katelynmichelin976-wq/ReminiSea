# 用户管理（注册/找回密码/改密） Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Spec:** `docs/superpowers/specs/2026-06-09-user-mgmt-design.md`

**Goal:** 在忆海拾光 PWA 账户屏加入注册新账号、邮件找回密码、登录后改密三套流程，全部使用 Supabase Auth 内置原语实现。

**Architecture:** 单文件 PWA `index.html` 内追加 3 个底部 sheet + 1 个全屏 reset-password screen + 4 个处理函数 + hash 路由（识别 `#/email-confirmed` 与 `#/reset-password`）。复用现有 `.account-field` / `.account-btn` / settings-overlay sheet 视觉规范。i18n 5 locale 同步加 ~30 keys。零后端代码。

**Tech Stack:** Vanilla JS、Supabase Auth JS SDK（`signUp` / `resetPasswordForEmail` / `updateUser` / `signInWithPassword`）、Playwright E2E、Windows + PowerShell。

```javascript
function openRegisterSheet() {
  const sheet = document.getElementById('register-sheet');
  if (!sheet) return;
  const emailEl = document.getElementById('account-email');
  if (emailEl) document.getElementById('reg-email').value = emailEl.value;
  document.getElementById('reg-pwd').value = '';
  document.getElementById('reg-pwd2').value = '';
  document.getElementById('reg-msg').style.display = 'none';
  document.getElementById('reg-pending-block').style.display = 'none';
  sheet.classList.add('open');
}
function closeRegisterSheet(e) {
  if (e && e.target !== document.getElementById('register-sheet')) return;
  document.getElementById('register-sheet').classList.remove('open');
}

async function doRegister() {
  const email = document.getElementById('reg-email').value.trim();
  const pwd   = document.getElementById('reg-pwd').value;
  const pwd2  = document.getElementById('reg-pwd2').value;
  const msgEl = document.getElementById('reg-msg');
  const btnEl = document.getElementById('reg-submit-btn');

  // validate
  if (!email || !/^[^@\s]+@[^@\s]+\.[^@\s]+$/.test(email)) {
    msgEl.style.display=''; msgEl.style.color='#9B3A2A'; msgEl.textContent = t('auth_email_invalid'); return;
  }
  if (pwd.length < 6) {
    msgEl.style.display=''; msgEl.style.color='#9B3A2A'; msgEl.textContent = t('auth_pwd_too_short'); return;
  }
  if (pwd !== pwd2) {
    msgEl.style.display=''; msgEl.style.color='#9B3A2A'; msgEl.textContent = t('auth_pwd_mismatch'); return;
  }

  btnEl.disabled = true;
  btnEl.textContent = t('reg_btn_pending');
  msgEl.style.display = 'none';

  try {
    if (!_sb) _sb = _createSupabaseClient();
    const redirectTo = location.origin + location.pathname + '#/email-confirmed';
    const { error } = await _sb.auth.signUp({ email, password: pwd, options: { emailRedirectTo: redirectTo } });
    if (error) {
      if (error.message && error.message.toLowerCase().includes('already')) {
        msgEl.style.display=''; msgEl.style.color='#9B3A2A';
        msgEl.innerHTML = t('reg_already_registered') + ' <a class="account-link" onclick="closeRegisterSheet({target:document.getElementById(\'register-sheet\')})">' + t('reg_goto_login') + '</a>';
      } else {
        msgEl.style.display=''; msgEl.style.color='#9B3A2A';
        msgEl.textContent = t('auth_send_fail') + ': ' + error.message;
      }
      return;
    }
    // success → show pending block
    document.querySelector('#register-sheet .sheet-title').style.display = 'none';
    document.getElementById('reg-email').style.display = 'none';
    document.getElementById('reg-pwd').style.display = 'none';
    document.getElementById('reg-pwd2').style.display = 'none';
    document.getElementById('reg-submit-btn').style.display = 'none';
    document.getElementById('reg-wait-body').textContent = t('reg_wait_body', { email });
    document.getElementById('reg-pending-block').style.display = '';
  } finally {
    btnEl.disabled = false;
    btnEl.textContent = t('reg_btn');
  }
}

async function doResendVerification() {
  const email = document.getElementById('reg-email').value.trim();
  if (!email || !_sb) return;
  const redirectTo = location.origin + location.pathname + '#/email-confirmed';
  await _sb.auth.resend({ type: 'signup', email, options: { emailRedirectTo: redirectTo } }).catch(() => {});
  showCloudToast(t('reg_wait_title'));
}
```

```javascript
function openChangePasswordSheet() {
  const sheet = document.getElementById('change-password-sheet');
  if (!sheet) return;
  document.getElementById('cp-old').value = '';
  document.getElementById('cp-new').value = '';
  document.getElementById('cp-new2').value = '';
  document.getElementById('cp-msg').style.display = 'none';
  sheet.classList.add('open');
}
function closeChangePasswordSheet(e) {
  if (e && e.target !== document.getElementById('change-password-sheet')) return;
  document.getElementById('change-password-sheet').classList.remove('open');

async function doChangePassword() {
  const oldPwd = document.getElementById('cp-old').value;
  const newPwd = document.getElementById('cp-new').value;
  const newPwd2 = document.getElementById('cp-new2').value;
  const msgEl = document.getElementById('cp-msg');
  const btnEl = document.getElementById('cp-submit-btn');

  if (newPwd.length < 6) {
    msgEl.style.display=''; msgEl.style.color='#9B3A2A'; msgEl.textContent = t('auth_pwd_too_short'); return;
  }
  if (newPwd !== newPwd2) {
    msgEl.style.display=''; msgEl.style.color='#9B3A2A'; msgEl.textContent = t('auth_pwd_mismatch'); return;
  }
  if (!_sb || !_cloudUserEmail) {
    msgEl.style.display=''; msgEl.style.color='#9B3A2A'; msgEl.textContent = t('sync_offline'); return;
  }

  btnEl.disabled = true;
  btnEl.textContent = t('reg_btn_pending');
  try {
    // Step 1: verify old password
    const verify = await _sb.auth.signInWithPassword({ email: _cloudUserEmail, password: oldPwd });
    if (verify.error) {
      msgEl.style.display=''; msgEl.style.color='#9B3A2A'; msgEl.textContent = t('change_pwd_old_wrong'); return;
    }
    // Step 2: update password
    const upd = await _sb.auth.updateUser({ password: newPwd });
    if (upd.error) {
      msgEl.style.display=''; msgEl.style.color='#9B3A2A'; msgEl.textContent = t('auth_send_fail') + ': ' + upd.error.message; return;
    }
    showCloudToast(t('change_pwd_done_toast'));
    closeChangePasswordSheet({ target: document.getElementById('change-password-sheet') });
  } finally {
    btnEl.disabled = false;
    btnEl.textContent = t('reset_set_btn');
  }
}
```

```javascript
function handleAuthHashRoute() {
  const h = location.hash || '';
  if (h.startsWith('#/email-confirmed')) {
    // Supabase 默认验证完成后 hash 包含 access_token；用户已自动登录
    location.hash = '';
    showCloudToast(t('auth_email_confirmed_toast'));
    const emailLast = localStorage.getItem('yihaiLastCloudEmail') || '';
    const emailEl = document.getElementById('account-email');
    if (emailEl && emailLast) emailEl.value = emailLast;
    showAccount();
    return;
  }
  if (h.startsWith('#/reset-password')) {
    // Supabase 把 access_token 放在 hash；SDK 已自动建立 recovery session
    location.hash = '';
    showScreen('screen-reset-password');
    return;
  }
}
// 在现有 restoreCloudSession 之后调用：
setTimeout(handleAuthHashRoute, 200);  // 等 SDK 处理完 hash
```
