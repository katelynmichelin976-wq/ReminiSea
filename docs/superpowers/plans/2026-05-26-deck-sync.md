# 牌组云端同步 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 统一云端 schema（`decks` + `deck_cards`），支持个人牌组跨设备同步，并清理本地 key 前缀问题。

**Architecture:** 新建 `decks`/`deck_cards` 两张表替代现有的 `server_decks`/`cards_pool`/`server_deck_cards` 三张表；本地 deck key 直接使用 UUID，不再加 `cloud_` 前缀；个人牌组在 `saveDeck`/`deleteDeck` 时静默推送云端，session 就绪后检查是否有新发布自动拉取。所有改动在单文件 `yihai_v5.1.html` 中完成。

**Tech Stack:** Supabase (PostgreSQL + RLS), vanilla JS, IndexedDB, `crypto.randomUUID()`, SQL migration file.

```sql
-- ── 2. deck_cards 表 ─────────────────────────────────────────────────
CREATE TABLE IF NOT EXISTS deck_cards (
  id         BIGINT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
  deck_id    TEXT NOT NULL REFERENCES decks(id) ON DELETE CASCADE,
  card_id    TEXT NOT NULL,
  name       TEXT NOT NULL,
  image_url  TEXT,
  audio_url  TEXT,
  sort_order INTEGER DEFAULT 0,
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX IF NOT EXISTS idx_deck_cards_deck_id ON deck_cards(deck_id);

-- ── 3. RLS ────────────────────────────────────────────────────────────
ALTER TABLE decks ENABLE ROW LEVEL SECURITY;

CREATE POLICY "decks_read" ON decks FOR SELECT
  USING (
    deck_type IN ('preset', 'shared')
    OR user_id = auth.uid()
  );

CREATE POLICY "decks_write" ON decks FOR ALL
  USING (user_id = auth.uid())
  WITH CHECK (user_id = auth.uid());

ALTER TABLE deck_cards ENABLE ROW LEVEL SECURITY;

CREATE POLICY "deck_cards_read" ON deck_cards FOR SELECT
  USING (
    EXISTS (
      SELECT 1 FROM decks
      WHERE id = deck_id
        AND (deck_type IN ('preset', 'shared') OR user_id = auth.uid())

  );

CREATE POLICY "deck_cards_write" ON deck_cards FOR ALL
  USING (
```

```javascript
async function migrateMediaKeys(oldIdx) {
  const db = await openDB().catch(() => null);
  if (!db) return;
  for (const m of oldIdx) {
    if (!m.key || !m.key.startsWith('cloud_')) continue;
    const newKey = m.key.slice(6);
    const tx = db.transaction('blobs', 'readwrite');
    const store = tx.objectStore('blobs');
    const allKeys = await new Promise(res => {
      const req = store.getAllKeys(); req.onsuccess = e => res(e.target.result);
    });
    for (const k of allKeys) {
      if (typeof k === 'string' && k.startsWith(m.key + '_')) {
        const newK = newKey + k.slice(m.key.length);
        const blob = await new Promise(res => { const r = store.get(k); r.onsuccess = e => res(e.target.result); });
        if (blob) { store.put(blob, newK); store.delete(k); }
      }
    }
  }
}
```

```javascript
async function uploadDeckToCloud(key) {
  if (!_syncEnabled || !_sb || !_cloudUserId) return;
  const meta = DECKS_META.find(m => m.key === key);
  if (!meta || meta.deck_type !== 'personal') return;
  const cards = DECKS[key] || [];
  try {
    // upsert deck row (不更新 updated_at — 草稿不触发学习者自动同步)
    await _sb.from('decks').upsert({
      id: key,
      user_id: _cloudUserId,
      name: meta.name,
      deck_type: 'personal',
      card_count: cards.length
    }, { onConflict: 'id', ignoreDuplicates: false });

    // replace all deck_cards
    await _sb.from('deck_cards').delete().eq('deck_id', key);
    if (cards.length > 0) {
      const rows = cards.map((c, i) => ({
        deck_id: key,
        card_id: c.id,
        name: c.name,
        image_url: c._imgUrl || null,
        audio_url: c._audUrl || null,
        sort_order: i
      }));
      // batch insert (max 500 rows per call)
      for (let b = 0; b < rows.length; b += 500) {
        await _sb.from('deck_cards').insert(rows.slice(b, b + 500));
      }
    }
    console.log('[cloud] uploadDeckToCloud ok:', key, cards.length, 'cards');
  } catch(e) {
    console.warn('[cloud] uploadDeckToCloud fail:', e.message);
  }
}
```

```javascript
async function checkPersonalDeckUpdates() {
  if (!_syncEnabled || !_sb || !_cloudUserId) return;
```

```javascript
async function downloadPersonalDeckFromCloud(deckId, deckName) {
  if (!_syncEnabled || !_sb) return;
  try {
    const { data: cards, error } = await _sb.from('deck_cards')
      .select('card_id,name,image_url,audio_url')
      .eq('deck_id', deckId)
      .order('sort_order');
    if (error || !cards) return;
    if (!cards.length) return;

    const syncNow = new Date().toISOString();
    const deckCards = [];
    await parallelMapLimit(cards, 3, async (c, i) => {
      const card = { id: c.card_id, name: c.name, img: '', audioUrl: '', details: [] };
      const tasks = [];
      if (c.image_url) {
        tasks.push(
          downloadMediaFromCDNorSupabase(c.image_url)
            .then(async blob => {
              card.img = URL.createObjectURL(blob);
              await saveMedia(`${deckId}_${card.id}_img`, blob).catch(() => {});
            }).catch(e => console.warn('[cloud] img failed:', e.message))
        );
      }
      if (c.audio_url) {
        tasks.push(
          downloadMediaFromCDNorSupabase(c.audio_url, audioMimeFromUrl(c.audio_url))
            .then(async blob => {
              card.audioUrl = URL.createObjectURL(blob);
              await saveMedia(`${deckId}_${card.id}_aud`, blob).catch(() => {});
            }).catch(e => console.warn('[cloud] aud failed:', e.message))
        );
      }
      await Promise.all(tasks);

    });

    DECKS[deckId] = deckCards;
    const existingIdx = DECKS_META.findIndex(m => m.key === deckId);
    if (existingIdx >= 0) DECKS_META[existingIdx] = { key: deckId, name: deckName, deck_type: 'personal' };
    else DECKS_META.push({ key: deckId, name: deckName, deck_type: 'personal' });
    saveDeckIndex();
    saveDeckCards(deckId, deckCards);
    renderDeckList();
    console.log('[cloud] downloadPersonalDeckFromCloud ok:', deckId, deckCards.length, 'cards');
  } catch(e) {
    console.warn('[cloud] downloadPersonalDeckFromCloud fail:', e.message);
  }
}
```

```javascript
async function publishDeck(key) {
  if (!_syncEnabled || !_sb || !_cloudUserId) {
    showCloudToast(t('sync_offline'), true); return;
  }
  const meta = DECKS_META.find(m => m.key === key);
  if (!meta || meta.deck_type !== 'personal') return;

  try {
    // 先推送最新草稿内容
    await uploadDeckToCloud(key);

    // 更新 updated_at = NOW()，触发其他设备自动拉取
    const { error } = await _sb.from('decks')
      .update({ updated_at: new Date().toISOString() })
      .eq('id', key)
      .eq('user_id', _cloudUserId);
    if (error) throw error;

    // 更新本地 sync_at，避免自己在下次 check 时再次下载
    localStorage.setItem('yihai_sync_at_' + key, new Date().toISOString());

    showCloudToast(t('deck_published_ok', { name: esc(meta.name) }));
    console.log('[cloud] publishDeck ok:', key);
  } catch(e) {
    showCloudToast(t('deck_publish_fail', { msg: e.message }), true);
    console.error('[cloud] publishDeck fail:', e);
  }
}
```

## 注意事项

1. **管理员 UUID 必须填写**：Migration 010 Step 2 必须将 `<admin_user_uuid>` 替换为实际值，否则 preset 牌组无 `user_id` 导致 RLS 查询失败。

2. **`server_deck_cards` 实际表名**：CLAUDE.md 中写作 `srv_deck_crd`，代码中用 `server_deck_cards`。Task 2 Step 11 改完后，确认 `server_deck_cards` 确实是旧表名，否则调整。

3. **旧三张表保留不删**：`server_decks`、`cards_pool`、`server_deck_cards` 在 migration 010 中不删除，待 Task 5 回归测试通过并稳定运行一段时间后再清理。

4. **`migrateMediaKeys` 是异步后台操作**：首次加载时 IndexedDB key 迁移在 `setTimeout(() => ..., 0)` 中执行，不阻塞主流程；迁移完成前媒体无法加载，但重启 app 后即恢复（迁移已完成）。

5. **`uploadDeckToCloud` 使用全量 replace**：先 delete 再 insert，对大牌组（>500 卡）效率低，但当前牌组规模（<200 卡）完全可接受。未来可改增量 upsert。
